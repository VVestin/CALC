Program->
	[COLON](Statement)(Program)

Statement->
	(ControlStructure) |
	(Expression)       |
	(Assignment)
	
Assignment->
	(Expression)[OPERATOR ->](Variable)
	
ControlStructure->
	[KEYWORD for](Expression)(Variable)[KEYWORD comma](Expression)[KEYWORD comma](Expression)(Program)[COLON][KEYWORD end] |
	[KEYWORD for](Expression)(Variable)[KEYWORD comma](Expression)[KEYWORD comma](Expression)[KEYWORD comma](Expression)(Program)[COLON][KEYWORD end] |
	[KEYWORD if](Expression)(Program)[COLON][KEYWORD end] |
	[KEYWORD if](Expression)(Program)[COLON][KEYWORD else](Program)[COLON][KEYWORD end] |
	[KEYWORD while](Expression)(Program)[COLON][KEYWORD end] |
	[KEYWORD repeat](Expression)(Program)[COLON][KEYWORD end] | 
	[KEYWORD Lbl](LabelName) |
	[KEYWORD Goto](LabelName)
	
LabelName->
	[INT_VAR] |
	[INT_VAR][INT_VAR] |
	[INT_VAR][INT_VAR][INT_VAR]
	
Variable->
	[FIXED_VAR]  |
	[INT_VAR]    |
	[STRING_VAR] |
	[LIST_VAR]   |
	[MATRIX_VAR]
