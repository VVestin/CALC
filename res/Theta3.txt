Fun LEFT(S,T,H)
For(Y,T,H)
Output(Y,S,Str1)
End
Return(S+1)
End

"*"->Str1
12->W
8->H
1->S
1->T
While(S<W and T<H)
For(X,S,W)
Output(T,X,Str1)
End
T+1->T
For(Y,T,H)
Output(Y,W,Str1)
End
W-1->W
For(X,S,W)
Output(H,X,Str1)
End
H-1->H
?For(Y,T,H)
?Output(Y,S,Str1)
?End
?S+1->S
LEFT(S,T,H)->S
Pause
End
